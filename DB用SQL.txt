◆初めに入力するもの◆
CREATE DATABASE STUDENTDB;
USE STUDENTDB;

CREATE TABLE STUDENT(
	STUDENTNO INT(8) PRIMARY KEY,
	STUDENTCLASS VARCHAR(20),
	STUDENTNAME VARCHAR(50),
	STUDENTADDYEAR INT(4),
	STUDENTSTOPCOUNT INT(1),
	STUDENTYEAR(1)
	);

◆適宜使用するもの◆
テーブル表示時
SELECT * FROM STUDENT;

テーブルソート表示時
SELECT * FROM STUDENT ORDER BY 属性;
//属性はウェブページのボタンそれぞれに設定され、ボタンクリック時にソート実行？

学生検索・番号検索時
SELECT * FROM STUDENT WHERE STUDENTNO = 学生番号;

学生検索・名前検索時
SELECT * FROM STUDENT WHERE STUDENTNAME = '学生氏名';

学生検索・所属クラス検索時
SELECT * FROM STUDENT WHERE STUDENTCLASS = '学生所属クラス';
//学生所属クラスはプルダウンリストから選ぶ(できれば)

学生追加時
INSERT INTO STUDENT VALUES(STUDENTNO,STUDENTCLASS,STUDENTNAME,NULL,STUDENTSTOPCOUNT,NULL);
UPDATE STUDENT SET STUDENTADDYEAR = 
	(SELECT SUBSTRING('STUDENT.STUDENTNO',1,4)), STUDENTYEAR = (現在年度 - STUDENTADDYEAR - STUDENTSTOPCOUNT) 
	WHERE STUDENTNO = 直前に入力した学生番号;

学生削除時
DELETE FROM STUDENT WHERE STUDENTNO = '入力データ';
//ウェブページ上の削除チェックボックスで指定された学生のIDが入力される

学生データ編集時
UPDATE STUDENT SET 
	STUDENTNO = 入力データ,
	STUDENTCLASS = '入力データ',
	STUDENTNAME = '入力データ',
	STUDENTSTOPCOUNT = 入力データ;
